def validate_tweet_user_tags(tweet_list)
  valid = []
  tagged = []
  tweet_list.each {|tweet| tagged = tagged | tweet.text.scan(/@[a-z0-9_-]+/)}
  tagged.each {|user| valid << user if @@user_model.user_exists?(user[/[^@]+/])}
  return valid.uniq
end
